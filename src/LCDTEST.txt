#include <TFT_eSPI.h>
#include <SPI.h>

TFT_eSPI tft = TFT_eSPI();

void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println("Serial communication started."); 
  delay(3000);
  Serial.println("before init");
  tft.init();
  Serial.println("after init");
  tft.setRotation(3); // 역 가로 방향
  tft.fillScreen(TFT_BLACK);
  tft.setTextColor(TFT_WHITE, TFT_BLACK);
  tft.setTextSize(2);
  tft.setCursor(10, 10);
  tft.println("Temp: 24.5 C");
  tft.setCursor(10, 40);
  tft.println("Humi: 60 %");
  tft.setCursor(10,70);
  tft.println("Co2 : 400ppm");
}

void loop() {
  // 센서 데이터 읽고 주기적으로 갱신하는 코드 추가 예정
}
